var paintBrush = {
    x: 100,
    y: 100,
    img: getImage("avatars/mr-pink")
};

var paintCanvas = function() {
    imageMode(CENTER);
    image(paintBrush.img, paintBrush.x, paintBrush.y);
};

paintCanvas();


mouseMoved = function() {
    //change x and y MauseX and mouseY
    paintBrush.x = mouseX;
    paintBrush.y = mouseY;
    paintCanvas();
};
